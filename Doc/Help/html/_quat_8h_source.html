244  inline Quat log() const
245  {
246  if ((x == 0) && (y == 0) && (z == 0))
247  {
248  if (w > 0)
249  {
250  return Quat(0, 0, 0, ::logf(w));
251  }
252  else if (w < 0)
253  {
254  // Log of a negative number. Multivalued, any number of the form
255  // (PI * v, ln(-q.w))
256  return Quat((float)G3D_PI, 0, 0, ::logf(-w));
257  }
258  else
259  {
260  // log of zero!
261  return Quat((float)nan(), (float)nan(), (float)nan(), (float)nan());
262  }
263  }
264  else
265  {
266  // Partly imaginary.
267  float imagLen = sqrtf(x * x + y * y + z * z);
268  float len = sqrtf(imagLen * imagLen + w * w);
269  float theta = atan2f(imagLen, (float)w);
270  float t = theta / imagLen;
271  return Quat(t * x, t * y, t * z, ::logf(len));
272  }
273  }
